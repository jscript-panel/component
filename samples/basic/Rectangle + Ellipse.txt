var colour = RGB(255, 0, 0);

function on_paint(gr) {
	gr.DrawRectangle(0, 0, 200, 200, 1, colour);
	gr.DrawEllipse(100, 100, 50, 50, 1, colour);
	gr.FillEllipse(300, 100, 50, 50, colour);
	gr.DrawRoundedRectangle(400, 0, 300, 200, 50, 50, 1, colour);
	gr.FillRoundedRectangle(450, 50, 200, 100, 50, 50, colour);
}
